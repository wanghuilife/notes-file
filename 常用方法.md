## 一、时间处理

### 1、根据天数返回日期范围

```js
/** 方法说明
 * @method dateRange
 * @description 根据天数计算以当前日期为基准计算开始日期
 * @param { Number-Int } dayNumber 最近日期天数
 * @return { Object} 返回日期范围
 * @author 大仙儿
 */

function dateRange(dayNumber) {
    const day = 24 * 60 * 60 * 1000
    const currTime = new Date().getTime(); // 当前时间戳
    const start = currTime - day * dayNumber // 其实日期时间戳
    function dateFormat(timestamp) {
        const date=new Date(timestamp)
        const year = date.getFullYear();
        const mouth = (date.getMonth() + 1) < 10 ? '0' + (date.getMonth() + 1) : (date.getMonth() + 1)
        const day = date.getDate() < 10 ? '0' + date.getDate() : date.getDate()
        return year + '-'+ mouth + '-' + day
    }
    return {
        start: dateFormat(start),
        end: dateFormat(currTime)
    }
}
```

**示例**

```js
系统当前日期：2020年7月24日
dateRange(20)
结果：
{ 
    start: "2020-07-04", 
    end: "2020-07-24" 
}
```

### 2、日期格式化

```js
/** 方法说明
 * @method dateFormat
 * @description 返回日期的年|月|日 时|分|秒|时间戳
 * @param { String, Number } dateSource 日期 
 * @return { Object } 返回对象
 * @author 大仙儿
 */

function dateFormat(dateSource) {
	const date = new Date(dateSource)
	const year = date.getFullYear();
	const mouth = (date.getMonth() + 1) < 10 ? '0' + (date.getMonth() + 1) : (date.getMonth() + 1);
	const day = date.getDate() < 10 ? '0' + date.getDate() : date.getDate();
	const hours = date.getHours() < 10 ? '0' + date.getHours() : date.getHours();
	const minutes = date.getMinutes() < 10 ? '0' + date.getMinutes() : date.getMinutes();
	const seconds = date.getSeconds() < 10 ? '0' + date.getSeconds() : date.getSeconds();
	const timestamp = date.getTime();

	return {
		year,
		mouth,
		day,
		hours,
		minutes,
		seconds,
		timestamp
	}
}
```

**示例**

```
dateFormat('2020-07-29 10:20:08')
结果：
{ 
	year: 2020,
	mouth: "07",
	day: 29,
	hours: 10,
	minutes: 20,
	seconds: "08",
	timestamp: 1595989208000
}
```

### 4、随机生成指定范围指定个数的数字[不重复]

```js
/** 方法说明
 * @method getRandomNumber
 * @description 随机生成指定范围内的数字
 * @param { Number } min 最小值
 * @param { Number } max 最大值
 * @param { Number } number 个数
 * @return { Array } 指定个数的数组
 * @author 大仙儿
 */

function getRandomNumber(min,max,number){
	if(max-min < number) {
		return ['指定范围数字小于指定个数']
	}
	let backArray =[];
	for(let i=0;i<number*2;i++) {
		let value = Math.floor(Math.random() * (max - min) ) + min;
		if(!backArray.includes(value)) {
			backArray.push(value);
		}
		if(backArray.length===number) {
			break;
		}
	}
	return backArray
}
```

```js
getRandomNumber(100,200,20)
getRandomNumber(1,10,5)
getRandomNumber(1,5,10)
结果：

[2,5,8,3,1]
```

### 100、校验集合

```js
/** 方法说明
 * @method checkStr
 * @description 类型判断集合
 * @param { String }  str 判断的字符串
 * @param { String }  type 判断类型
 * @return { Boolean } true/false
 * @return-example  true
 * @author 大仙儿 
 */

function checkStr(str, type) {
    switch (type) {
        case 'phone':   //手机号码
            return /^1[3|4|5|6|7|8|9][0-9]{9}$/.test(str);
        case 'tel':     //座机
            return /^(0\d{2,3}-\d{7,8})(-\d{1,4})?$/.test(str);
        case 'card':    //身份证
            return /(^\d{15}$)|(^\d{18}$)|(^\d{17}(\d|X|x)$)/.test(str);
        case 'pwd':     //密码以字母开头，长度在6~18之间，只能包含字母、数字和下划线
            return /^[a-zA-Z]\w{5,17}$/.test(str)
        case 'postal':  //邮政编码
            return /[1-9]\d{5}(?!\d)/.test(str);
        case 'QQ':      //QQ号
            return /^[1-9][0-9]{4,9}$/.test(str);
        case 'email':   //邮箱
            return /^[\w-]+(\.[\w-]+)*@[\w-]+(\.[\w-]+)+$/.test(str);
        case 'money':   //金额(小数点2位)
            return /^\d*(?:\.\d{0,2})?$/.test(str);
        case 'URL':     //网址
            return /(http|ftp|https):\/\/[\w\-_]+(\.[\w\-_]+)+([\w\-\.,@?^=%&:/~\+#]*[\w\-\@?^=%&/~\+#])?/.test(str)
        case 'IP':      //IP
            return /((?:(?:25[0-5]|2[0-4]\\d|[01]?\\d?\\d)\\.){3}(?:25[0-5]|2[0-4]\\d|[01]?\\d?\\d))/.test(str);
        case 'date':    //日期时间
            return /^(\d{4})\-(\d{2})\-(\d{2}) (\d{2})(?:\:\d{2}|:(\d{2}):(\d{2}))$/.test(str) || /^(\d{4})\-(\d{2})\-(\d{2})$/.test(str)
        case 'number':  //数字
            return /^[0-9]$/.test(str);
        case 'english': //英文
            return /^[a-zA-Z]+$/.test(str);
        case 'chinese': //中文
            return /^[\\u4E00-\\u9FA5]+$/.test(str);
        case 'lower':   //小写
            return /^[a-z]+$/.test(str);
        case 'upper':   //大写
            return /^[A-Z]+$/.test(str);
        case 'HTML':    //HTML标记
            return /<("[^"]*"|'[^']*'|[^'">])*>/.test(str);
        default:
            return true;
    }
}
```

**示例**

```
checkStr("18200000000","phone")
结果：
true
```

```js
/** 方法说明
 * @method arraySort
 * @description 数组排序
 * @param { Array }  arr  排序数据源
 * @param { Number } type 排序方式  1：从小到大 2：从大到小 3：随机
 * @return { Array } [...]
 * @return-example
 * @author 大仙儿 
 */

function arraySort(arr, type = 1) {
    return arr.sort((a, b) => {
        switch (type) {
            case 1:
                return a - b;
            case 2:
                return b - a;
            case 3:
                return Math.random() - 0.5;
            default:
                return arr;
        }
    })
}
```

### 11、字符串大小写改变



```js
/** 方法说明
 * @method changeCase
 * @description 字母大小写转换
 * @param { String } str 需要转换的字符串
 * @param { Number } type: 1、首字母大写 2、首字母小写 3、大小写转换 4、全部大写 5、全部小写
 * @return { String } 转换后的字符串
 * @author 大仙儿 
 */

function changeCase(str, type = 4) {
    switch (type) {
        case 1:
            return str.replace(/\b\w+\b/g, function (word) {
                return word.substring(0, 1).toUpperCase() + word.substring(1).toLowerCase();

            });
        case 2:
            return str.replace(/\b\w+\b/g, function (word) {
                return word.substring(0, 1).toLowerCase() + word.substring(1).toUpperCase();
            });
        case 3:
            return str.split('').map(function (word) {
                if (/[a-z]/.test(word)) {
                    return word.toUpperCase();
                } else {
                    return word.toLowerCase()
                }
            }).join('')
        case 4:
            return str.toUpperCase();
        case 5:
            return str.toLowerCase();
        default:
            return str;
    }
}
```

**示例**

```js
let strSource = "DaXian"
changeCase(strSource, 1);
changeCase(strSource, 2);
changeCase(strSource, 3);
changeCase(strSource, 4);
changeCase(strSource, 5);
结果：
Daxian

DAXIAN
```

### 12、密码强度检查

```js
/** 方法说明
 * @method checkPasswordLevel
 * @description 检查密码等级
 * @param { String } str 密码
 * @return { Number } 密码等级
 * @author 大仙儿 
 */

function checkPasswordLevel(str) {
    var level = 0;
    if (str.length < 6) {
        return level
    }
    if (/[0-9]/.test(str)) {
        level++
    }
    if (/[a-z]/.test(str)) {
        level++
    }
    if (/[A-Z]/.test(str)) {
        level++
    }
    if (/[\.|-|_]/.test(str)) {
        level++
    }
    return level;
}
checkPasswordLevel('12345')
checkPasswordLevel('sdfsfdDDD3')
checkPasswordLevel('sdfasdf2344@@')

3
2
```
